// Elementos del DOM const taskInput = document.getElementById('taskInput'); const addTaskBtn = document.getElementById('addTaskBtn'); const taskList = document.getElementById('taskList'); // Evento para añadir al hacer click addTaskBtn.addEventListener('click', () => { const taskText = taskInput.value; if (taskText === '') return; crearElementoTarea(taskText); guardarTareas(); // Actualizamos el localstorage taskInput.value = ''; }); // Función para montar el LI y el botón de borrar function crearElementoTarea(texto) { const li = document.createElement('li'); li.textContent = texto; const btnBorrar = document.createElement('button'); btnBorrar.textContent = 'Eliminar'; btnBorrar.onclick = () => { li.remove(); guardarTareas(); }; li.appendChild(btnBorrar); taskList.appendChild(li); } // Para que se guarden los datos en JSON function guardarTareas() { const tareas = []; document.querySelectorAll('#taskList li').forEach(li => { tareas.push(li.firstChild.textContent); }); localStorage.setItem('misTareas', JSON.stringify(tareas)); } // Carga inicial al arrancar la web window.onload = () => { const recuperadas = JSON.parse(localStorage.getItem('misTareas')); if (recuperadas) { recuperadas.forEach(t => crearElementoTarea(t)); } }; window.onload = () => { const tareasGuardadas = JSON.parse(localStorage.getItem('misTareas')); if (tareasGuardadas) { tareasGuardadas.forEach(texto => crearElementoTarea(texto)); } };
